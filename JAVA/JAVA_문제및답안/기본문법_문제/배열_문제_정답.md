    int temp = 0;
    for(int i=1; i<array05.length; i++) {
        for(int j=0; j<array05.length-1; j++) {
            if(array05[j] > array05[i]) {
                temp = array05[i];
                array05[i] = array05[j];
                array05[j] = temp;
            }
        }
    }
